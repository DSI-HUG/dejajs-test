export const createCounterRangeValidator = (maxValue: number, minValue: number) => {
    return (c: FormControl) => {
        const err = {
            rangeError: {
                given: c.value,
                max: maxValue,
                min: minValue
            }
        };
 
        return ((maxValue && c.value > maxValue) || (minValue && c.value < minValue)) ? err : null;
    };
};
